{{/* Compute one description per page so meta/og/twitter stay in sync. Home → site description. Pages → explicit .Description, else a plainified, whitespace-collapsed ~160-char truncation of .Summary, else site description. */}} {{- $desc := .Site.Params.description -}} {{- if .IsPage -}} {{- if .Description -}} {{- $desc = .Description -}} {{- else if .Summary -}} {{- $desc = .Summary | plainify | transform.HTMLUnescape | replaceRE `\s+` " " | truncate 160 -}} {{- end -}} {{- end -}} {{/* Per-post OG image generated at build time via images.Text. Every content section (legacy /posts/ + the six reader-profile sections) gets its own card; other pages (home, archive, taxonomies) fall back to the static masthead at /og-image.png. */}} {{- $ogImage := printf "%sog-image.png" .Site.BaseURL -}} {{- $contentSections := slice "posts" -}} {{- range .Site.Params.reader_profiles -}} {{- $contentSections = $contentSections | append .slug -}} {{- end -}} {{- if and .IsPage (in $contentSections .Section) -}} {{- $base := resources.Get "og/base.png" -}} {{- $font := resources.Get "fonts/og-title.ttf" -}} {{- if and $base $font -}} {{/* Word-wrap the title into lines of ≤ ~18 chars by word. */}} {{- $title := .Title -}} {{- $words := split $title " " -}} {{- $lines := slice -}} {{- $current := "" -}} {{- range $words -}} {{- $candidate := cond (eq $current "") . (printf "%s %s" $current .) -}} {{- if le (len $candidate) 18 -}} {{- $current = $candidate -}} {{- else -}} {{- if ne $current "" -}} {{- $lines = $lines | append $current -}} {{- end -}} {{- $current = . -}} {{- end -}} {{- end -}} {{- if ne $current "" -}} {{- $lines = $lines | append $current -}} {{- end -}} {{/* Cap to 3 lines so long titles never overflow the canvas. */}} {{- if gt (len $lines) 3 -}} {{- $lines = first 3 $lines -}} {{- end -}} {{- $wrapped := delimit $lines "\n" -}} {{/* Scale title down when it wraps to 3 lines so the bottom rule isn't crossed. 1-2 line titles stay large and dramatic. */}} {{- $size := 92 -}} {{- $y := 240 -}} {{- if ge (len $lines) 3 -}} {{- $size = 70 -}} {{- $y = 220 -}} {{- end -}} {{- $opts := dict "color" "#1a1a1a" "size" $size "linespacing" 14 "x" 90 "y" $y "font" $font -}} {{- $generated := $base.Filter (images.Text $wrapped $opts) -}} {{- $ogImage = $generated.Permalink -}} {{- end -}} {{- end -}} {{/* Stash for single.html → JSON-LD BlogPosting (avoids re-computing). */}} {{- .Store.Set "metaDescription" $desc -}} {{- .Store.Set "ogImageURL" $ogImage -}} {{ if .IsHome }}{{ .Site.Title }} — {{ .Site.Params.description }}{{ else }}{{ .Title }} — {{ .Site.Title }}{{ end }} {{ with .OutputFormats.Get "RSS" }} {{ end }}
