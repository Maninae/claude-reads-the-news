{{ define "main" }} {{/* Homepage: "today across all tabs" overview. Registry-order card grid of the latest entry per profile (profiles with zero entries are simply skipped), followed by an interleaved "Earlier" stream of every profile entry, newest first (excluding the cards above). Legacy content/posts/ entries are NOT shown here. They stay reachable from the archive. */}} {{- $profiles := .Site.Params.reader_profiles -}} {{/* Latest entry per profile, in registry order, skipping empty profiles. $latestIDs is the RelPermalink set we exclude from the "Earlier" stream. */}} {{- $latestPerProfile := slice -}} {{- $latestIDs := slice -}} {{- range $profiles -}} {{- $slug := .slug -}} {{- $label := .label -}} {{- $sectionPages := where $.Site.RegularPages "Section" $slug -}} {{- $ordered := $sectionPages.ByDate.Reverse -}} {{- with index $ordered 0 -}} {{- $latestPerProfile = $latestPerProfile | append (dict "page" . "profile" $slug "label" $label) -}} {{- $latestIDs = $latestIDs | append .RelPermalink -}} {{- end -}} {{- end -}} {{/* Today-strip: one card per profile that has any entries. */}} {{- if $latestPerProfile }}
{{- range $latestPerProfile }} {{- $p := .page }}
{{ .label }}

{{ $p.Title }}

{{ with $p.Params.description }}{{ . }}{{ else }}{{ $p.Plain | transform.HTMLUnescape | truncate 140 }}{{ end }}

{{ $p.Date.Format "Jan 2" | upper }}
{{- end }}
{{- end }} {{/* "Earlier" stream: every profile entry across the six sections, newest first, minus the cards above. Reuses the same server-rendered grid shape so feed.js batching works unchanged. Section-label chip (not the entry's topic) so every cell on the homepage announces which desk it came from, consistent with the today-cards and the archive rows. Section pages themselves keep the topic chip since the desk is already the page context. */}} {{- $profileSlugs := slice -}} {{- $slugToLabel := dict -}} {{- range $profiles -}} {{- $profileSlugs = $profileSlugs | append .slug -}} {{- $slugToLabel = merge $slugToLabel (dict .slug .label) -}} {{- end -}} {{- $allProfileEntries := where .Site.RegularPages "Section" "in" $profileSlugs -}} {{- $allOrdered := $allProfileEntries.ByDate.Reverse -}} {{- $earlier := slice -}} {{- range $allOrdered -}} {{- if not (in $latestIDs .RelPermalink) -}} {{- $earlier = $earlier | append . -}} {{- end -}} {{- end -}} {{- if $earlier }}
{{- range $earlier }}
{{ index $slugToLabel .Section }}
